```typescript
// Simplified orchestration concept
interface AgentWorkflow {
  pattern: DeployablePattern;
  steps: AgentStep[];
  providers: AIProvider[];
}

type AIProvider =
  | 'claude-code'
  | 'codex'
  | 'chatgpt'
  | 'gemini'
  | 'cursor'
  | 'kilo';

interface AgentStep {
  agent: AIProvider;
  task: string;
  input: any;
  output: any;
  nextStep?: string;
}

// Example: RAG System Build
const ragWorkflow: AgentWorkflow = {
  pattern: ragSystemPattern,
  steps: [
    {
      agent: 'claude-code',
      task: 'Generate base RAG implementation',
      input: { template: 'patterns/rag-system' },
      output: { files: [...] },
      nextStep: 'optimize-embeddings'
    },
    {
      agent: 'codex',
      task: 'Optimize embedding generation',
      input: { files: [...] },
      output: { optimizedFiles: [...] },
      nextStep: 'add-ui'
    },
    {
      agent: 'cursor',
      task: 'Build chat UI',
      input: { optimizedFiles: [...] },
      output: { fullApp: [...] },
      nextStep: 'test'
    },
    {
      agent: 'kilo',
      task: 'Run integration tests',
      input: { fullApp: [...] },
      output: { testResults: {...} },
      nextStep: 'deploy'
    }
  ],
  providers: ['claude-code', 'codex', 'cursor', 'kilo']
```
